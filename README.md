            <div style="display:flex;justify-content:space-between;gap:8px;align-items:center;">
              <div>
                <a href="${d.url}">${d.title}</a>
                <div class="sig-tiny" style="margin-top:4px;">${d.description}</div>
              </div>
              <div>${(d.tags||[]).map(t=>`<span class="sig-tag">#${t}</span>`).join("")}</div>
            </div>
          `;
          list.appendChild(li);
        });
        status.innerText = `Found ${items.length} result(s).`;
      };

      const run = () => render(search(qEl.value));
      btn.addEventListener("click", run);
      qEl.addEventListener("input", run);
    })();

    // =========================
    // Featured Video
    // =========================
    (function(){
      const iframe = document.getElementById("sig-featured-iframe");
      const note = document.getElementById("sig-featured-note");
      if (!iframe || !note) return;
      iframe.src = ytNoCookie(DATA.featuredVideo.id);
      note.innerText = DATA.featuredVideo.title + " — " + DATA.featuredVideo.note;
    })();

    // =========================
    // Playlist + timestamps + notes
    // =========================
    (function(){
      const list = document.getElementById("sig-video-list");
      const tsList = document.getElementById("sig-timestamps");
      const notes = document.getElementById("sig-notes");
      const featured = document.getElementById("sig-featured-iframe");
      if (!list || !tsList || !notes || !featured) return;

      let active = DATA.videos[0]?.id || null;

      const renderPlaylist = () => {
        list.innerHTML = "";
        DATA.videos.forEach(v => {
          const li = document.createElement("li");
          li.innerHTML = `
            <div style="display:flex;justify-content:space-between;gap:8px;align-items:center;">
              <div>
                <a href="${v.url}" target="_blank" rel="noopener">${v.title}</a>
                <div class="sig-tiny" style="margin-top:4px;">${(v.tags||[]).map(t=>`<span class="sig-tag">#${t}</span>`).join("")}</div>
              </div>
              <button class="sig-btn" data-id="${v.id}">${active===v.id ? "Playing" : "Play"}</button>
            </div>
          `;
          li.querySelector("button").addEventListener("click", () => setActive(v.id));
          list.appendChild(li);
        });
      };

      const renderTimestamps = id => {
        tsList.innerHTML = "";
        const v = DATA.videos.find(x => x.id === id);
        if (!v || !v.timestamps || !v.timestamps.length) {
          tsList.innerHTML = `<li class="sig-tiny">No timestamps yet.</li>`;
          return;
        }
        v.timestamps.forEach(ts => {
          const li = document.createElement("li");
          li.innerHTML = `
            <div class="sig-row">
              <button class="sig-btn" data-ts="${ts.t}">${ts.t}</button>
              <span>${ts.label}</span>
            </div>
          `;
          li.querySelector("button").addEventListener("click", () => {
            const s = mmssToSeconds(ts.t);
            if (s != null) featured.src = ytNoCookie(id) + "&start=" + s;
          });
          tsList.appendChild(li);
        });
      };

      const renderNotes = id => {
        const v = DATA.videos.find(x => x.id === id);
        notes.innerText = v?.notes || "Add editorial notes in DATA.videos[].notes to brief your viewers.";
      };

      const setActive = id => {
        active = id;
        featured.src = ytNoCookie(id);
        renderPlaylist();
        renderTimestamps(id);
        renderNotes(id);
      };

      // Initial render
      renderPlaylist();
      setActive(active);
    })();

    // =========================
    // Clip Generator
    // =========================
    (function(){
      const titleEl = document.getElementById("sig-clip-title");
      const startEl = document.getElementById("sig-clip-start");
      const endEl = document.getElementById("sig-clip-end");
      const makeBtn = document.getElementById("sig-clip-make");
      const urlEl = document.getElementById("sig-clip-url");
      const copyBtn = document.getElementById("sig-clip-copy");
      const featured = document.getElementById("sig-featured-iframe");
      if (!titleEl || !startEl || !endEl || !makeBtn || !urlEl || !copyBtn || !featured) return;

      const getActiveId = () => {
        const src = featured.src || "";
        const m = src.match(/embed\/([^?]+)/);
        return m ? m[1] : DATA.featuredVideo.id;
      };

      makeBtn.addEventListener("click", () => {
        const id = getActiveId();
        const s = mmssToSeconds(startEl.value.trim());
        const e = mmssToSeconds(endEl.value.trim());
        if (s == null || e == null || e <= s) {
          urlEl.value = "Invalid times. Use mm:ss and ensure end > start.";
          return;
        }
        const base = `https://www.youtube-nocookie.com/embed/${id}?rel=0&start=${s}&end=${e}`;
        const t = titleEl.value.trim();
        urlEl.value = t ? `${base}#${encodeURIComponent(t)}` : base;
      });

      copyBtn.addEventListener("click", async () => {
        try {
          await navigator.clipboard.writeText(urlEl.value);
          copyBtn.innerText = "Copied!";
          setTimeout(()=> copyBtn.innerText="Copy", 1500);
        } catch (e) {
          copyBtn.innerText = "Copy failed";
          setTimeout(()=> copyBtn.innerText="Copy", 1500);
        }
      });
    })();
  </script>
